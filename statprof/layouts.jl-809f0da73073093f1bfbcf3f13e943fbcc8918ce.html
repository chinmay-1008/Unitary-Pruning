330 # recursively compute the bounding boxes for the layout and plotarea (relative to canvas!)
331 function update_child_bboxes!(layout::GridLayout, minimum_perimeter = [0mm, 0mm, 0mm, 0mm])
332 nr, nc = size(layout)
333
334 # create a matrix for each minimum padding direction
335 minpad_left, minpad_top, minpad_right, minpad_bottom = paddings(layout.grid)
336
337 # get the max horizontal (left and right) padding over columns,
338 # and max vertical (bottom and top) padding over rows
339 # TODO: add extra padding here
340 pad_left = maximum(minpad_left, dims = 1)
341 pad_top = maximum(minpad_top, dims = 2)
342 pad_right = maximum(minpad_right, dims = 1)
343 pad_bottom = maximum(minpad_bottom, dims = 2)
344
345 # make sure the perimeter match the parent
346 pad_left[1] = max(pad_left[1], leftpad(minimum_perimeter))
347 pad_top[1] = max(pad_top[1], toppad(minimum_perimeter))
348 pad_right[end] = max(pad_right[end], rightpad(minimum_perimeter))
349 pad_bottom[end] = max(pad_bottom[end], bottompad(minimum_perimeter))
350
351 # scale this up to the total padding in each direction, and limit padding to 95%
352 total_pad_horizontal = min(0.95width(layout), sum(pad_left + pad_right))
353 total_pad_vertical = min(0.95height(layout), sum(pad_top + pad_bottom))
354
355 # now we can compute the total plot area in each direction
356 total_plotarea_horizontal = width(layout) - total_pad_horizontal
357 total_plotarea_vertical = height(layout) - total_pad_vertical
358
359 @assert total_plotarea_horizontal > 0mm
360 @assert total_plotarea_vertical > 0mm
361
362 # recompute widths/heights
363 layout.widths = recompute_lengths(layout.widths)
364 layout.heights = recompute_lengths(layout.heights)
365
366 # we have all the data we need... lets compute the plot areas and set the bounding boxes
367 for r in 1:nr, c in 1:nc
368 child = layout[r, c]
369
370 # get the top-left corner of this child... the first one is top-left of the parent (i.e. layout)
371 child_left = c == 1 ? left(layout.bbox) : right(layout[r, c - 1].bbox)
372 child_top = r == 1 ? top(layout.bbox) : bottom(layout[r - 1, c].bbox)
373
374 # compute plot area
375 plotarea_left = child_left + pad_left[c]
376 plotarea_top = child_top + pad_top[r]
377 plotarea_width = total_plotarea_horizontal * layout.widths[c]
378 plotarea_height = total_plotarea_vertical * layout.heights[r]
379
380 bb = BoundingBox(plotarea_left, plotarea_top, plotarea_width, plotarea_height)
381 plotarea!(child, bb)
382
383 # compute child bbox
384 child_width = pad_left[c] + plotarea_width + pad_right[c]
385 child_height = pad_top[r] + plotarea_height + pad_bottom[r]
386 bbox!(child, BoundingBox(child_left, child_top, child_width, child_height))
387
388 # this is the minimum perimeter as decided by this child's parent, so that
389 # all children on this border have the same value
390 min_child_perim = [
391 c == 1 ? leftpad(layout) : pad_left[c],
392 r == 1 ? toppad(layout) : pad_top[r],
393 c == nc ? rightpad(layout) : pad_right[c],
394 r == nr ? bottompad(layout) : pad_bottom[r],
395 ]
396 # recursively update the child's children
397 update_child_bboxes!(child, min_child_perim)
398 end
399 end

632 link_axes!(layout.grid, :xaxis)
633 link_axes!(layout.grid, :yaxis)
634 end
635 foreach(l -> link_axes!(l, link), layout.grid)
636 end

640 function twin(sp, letter)
641 plt = sp.plt
642 orig_sp = first(plt.subplots)
643 for letter in filter(!=(letter), axes_letters(orig_sp, letter))
644 ax = orig_sp[get_attr_symbol(letter, :axis)]
645 ax[:grid] = false # disable the grid (overlaps with twin axis)
646 end
647 if orig_sp[:framestyle] === :box
648 # incompatible with shared axes (see github.com/JuliaPlots/Plots.jl/issues/2894)
649 orig_sp[:framestyle] = :axes
650 end
651 plot!(
652 plt;
653 inset = (sp[:subplot_index], bbox(0, 0, 1, 1)),
654 left_margin = orig_sp[:left_margin],
655 top_margin = orig_sp[:top_margin],
656 right_margin = orig_sp[:right_margin],
657 bottom_margin = orig_sp[:bottom_margin],
658 )
659 twin_sp = last(plt.subplots)
660 letters = axes_letters(twin_sp, letter)
661 tax, oax = map(l -> twin_sp[get_attr_symbol(l, :axis)], letters)
662 tax[:grid] = false
663 tax[:showaxis] = false
664 tax[:ticks] = :none
665 oax[:grid] = false
666 oax[:mirror] = true
667 twin_sp[:background_color_inside] = RGBA{Float64}(0, 0, 0, 0)
668 link_axes!(sp[get_attr_symbol(letter, :axis)], tax)
669 twin_sp
670 end
671
672 """
673 twinx(sp)
674
675 Adds a new, empty subplot overlaid on top of `sp`, with a mirrored y-axis and linked x-axis.
676 """
677 twinx(sp::Subplot) = twin(sp, :x)
678 twinx(plt::Plot = current()) = twinx(first(plt))

681 twiny(sp)
682
683 Adds a new, empty subplot overlaid on top of `sp`, with a mirrored x-axis and linked y-axis.
684 """
685 twiny(sp::Subplot) = twin(sp, :y)
686 twiny(plt::Plot = current()) = twiny(first(plt))
