988 """
989 `default(key)` returns the current default value for that key.
990
991 `default(key, value)` sets the current default value for that key.
992
993 `default(; kw...)` will set the current default value for each key/value pair.
994
995 `default(plotattributes, key)` returns the key from plotattributes if it exists, otherwise `default(key)`.
996
997 """
998 function default(k::Symbol)
999 k = get(_keyAliases, k, k)
1000 for defaults in _all_defaults
1001 haskey(defaults, k) && return defaults[k]
1002 end
1003 haskey(_axis_defaults, k) && return _axis_defaults[k]
1004 if (axis_k = parse_axis_kw(k)) !== nothing
1005 letter, key = axis_k
1006 return _axis_defaults_byletter[letter][key]
1007 end
1008 k === :letter && return k # for type recipe processing
1009 missing
1010 end
1011
1012 function default(k::Symbol, v)
1013 k = get(_keyAliases, k, k)
1014 for defaults in _all_defaults
1015 if haskey(defaults, k)
1016 defaults[k] = v
1017 return v
1018 end
1019 end
1020 if haskey(_axis_defaults, k)
1021 _axis_defaults[k] = v
1022 return v
1023 end
1024 if (axis_k = parse_axis_kw(k)) !== nothing
1025 letter, key = axis_k
1026 _axis_defaults_byletter[letter][key] = v
1027 return v
1028 end
1029 k in _suppress_warnings || error("Unknown key: ", k)
1030 end

1033 (reset && isempty(kw)) && reset_defaults()
1034 kw = KW(kw)
1035 Plots.preprocess_attributes!(kw)
1036 for (k, v) in kw
1037 default(k, v)
1038 end
1039 end
