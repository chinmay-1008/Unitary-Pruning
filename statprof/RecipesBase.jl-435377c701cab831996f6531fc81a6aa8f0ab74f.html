136 # process the body of the recipe recursively.
137 # when we see the series macro, we split that block off:
138 # let
139 # d2 = copy(d)
140 # <process_recipe_body on d2>
141 # RecipeData(d2, block_return)
142 # end
143 # and we push this block onto the series_blocks list.
144 # then at the end we push the main body onto the series list
145 function process_recipe_body!(expr::Expr)
146 for (i, e) in enumerate(expr.args)
147 if isa(e, Expr)
148 # process trailing flags, like:
149 # a --> b, :quiet, :force
150 quiet, require, force = false, false, false
151 if _is_arrow_tuple(e)
152 for flag in e.args
153 if _equals_symbol(flag, :quiet)
154 quiet = true
155 elseif _equals_symbol(flag, :require)
156 require = true
157 elseif _equals_symbol(flag, :force)
158 force = true
159 end
160 end
161 e = e.args[1]
162 end
163
164 # the unused operator `:=` will mean force: `x := 5` is equivalent to `x --> 5, force`
165 # note: this means "x is defined as 5"
166 if e.head === :(:=)
167 force = true
168 e.head = :(-->)
169 end
170
171 # we are going to recursively swap out `a --> b, flags...` commands
172 # note: this means "x may become 5"
173 if e.head === :(-->)
174 k, v = e.args
175 if isa(k, Symbol)
176 k = QuoteNode(k)
177 end
178
179 set_expr = if force
180 # forced override user settings
181 :(plotattributes[$k] = $v)
182 else
183 # if the user has set this keyword, use theirs
184 :($RecipesBase.is_explicit(plotattributes, $k) || (plotattributes[$k] = $v))
185 end
186
187 expr.args[i] = if quiet
188 # quietly ignore keywords which are not supported
189 :($RecipesBase.is_key_supported($k) ? $set_expr : nothing)
190 elseif require
191 # error when not supported by the backend
192 :(
193 $RecipesBase.is_key_supported($k) ? $set_expr :
194 error(
195 "In recipe: required keyword ",
196 $k,
197 " is not supported by backend $(backend_name())",
198 )
199 )
200 else
201 set_expr
202 end
203
204 elseif e.head ≡ :return
205 # To allow `return` in recipes just extract the returned arguments.
206 expr.args[i] = first(e.args)
207
208 elseif e.head ≢ :call
209 # we want to recursively replace the arrows, but not inside function calls
210 # as this might include things like Dict(1=>2)
211 process_recipe_body!(e)
212 end
213 end
214 end
215 end

219 """
220 This handy macro will process a function definition, replace `-->` commands, and
221 then add a new version of `RecipesBase.apply_recipe` for dispatching on the arguments.
222
223 This functionality is primarily geared to turning user types and settings into the
224 data and attributes that describe a Plots.jl visualization.
225
226 Set attributes using the `-->` command, and return a comma separated list of arguments that
227 should replace the current arguments.
228
229 An example:
230
231 ```julia
232 using RecipesBase
233
234 # Our custom type that we want to display
235 struct T end
236
237 @recipe function plot(t::T, n::Integer = 1; customcolor = :green)
238 markershape --> :auto, :require
239 markercolor --> customcolor, :force
240 xrotation --> 5
241 zrotation --> 6, :quiet
242 rand(10,n)
243 end
244
245 # ---------------------
246
247 # Plots will be the ultimate consumer of our recipe in this example
248 using Plots; gr()
249
250 # This call will implicitly call `RecipesBase.apply_recipe` as part of the Plots
251 # processing pipeline (see the Pipeline section of the Plots documentation).
252 # It will plot 5 line plots, all with black circles for markers.
253 # The markershape argument must be supported, and the zrotation argument's warning
254 # will be suppressed. The user can override all arguments except markercolor.
255 plot(T(), 5; customcolor = :black, shape=:c)
256 ```
257
258 In this example, we see lots of the machinery in action. We create a new type `T` which
259 we will use for dispatch, and an optional argument `n`, which will be used to determine the
260 number of series to display. User-defined keyword arguments are passed through, and the
261 `-->` command can be trailed by flags:
262
263 - quiet: Suppress unsupported keyword warnings
264 - require: Error if keyword is unsupported
265 - force: Don't allow user override for this keyword

500 function create_grid_vcat(expr::Expr)
501 rowsizes = map(rowsize, expr.args)
502 rmin, rmax = extrema(rowsizes)
503 if rmin > 0 && rmin == rmax
504 # we have a grid... build the whole thing
505 # note: rmin is the number of columns
506 nr = length(expr.args)
507 nc = rmin
508 body = Expr(:block)
509 for r in 1:nr
510 if (arg = expr.args[r]) |> isrow
511 for (c, item) in enumerate(arg.args)
512 push!(body.args, :(cell[$r, $c] = $(create_grid(item))))
513 end
514 else
515 push!(body.args, :(cell[$r, 1] = $(create_grid(arg))))
516 end
517 end
518 quote
519 let cell = Matrix(undef, $nr, $nc)
520 $body
521 cell
522 end
523 end
524 else
525 # otherwise just build one row at a time
526 sub(x) = :(cell[$(first(x)), 1] = $(create_grid(last(x))))
527 quote
528 let cell = Matrix(undef, $(length(expr.args)), 1)
529 $(map(sub, enumerate(expr.args))...)

532 end
