370 function get_minor_ticks(sp, axis, ticks_and_labels)
371 no_minor_intervals(axis) && return
372 ticks = first(ticks_and_labels)
373 length(ticks) < 2 && return
374
375 amin, amax = axis_limits(sp, axis[:letter])
376 scale = axis[:scale]
377 base = get(_logScaleBases, scale, nothing)
378
379 # add one phantom tick either side of the ticks to ensure minor ticks extend to the axis limits
380 if (log_scaled = scale _logScales)
381 sub = round(Int, log(base, ticks[2] / ticks[1]))
382 ticks = [ticks[1] / base; ticks; ticks[end] * base]
383 else
384 sub = 1 # unused
385 ratio = length(ticks) > 2 ? (ticks[3] - ticks[2]) / (ticks[2] - ticks[1]) : 1
386 first_step = ticks[2] - ticks[1]
387 last_step = ticks[end] - ticks[end - 1]
388 ticks = [ticks[1] - first_step / ratio; ticks; ticks[end] + last_step * ratio]
389 end
390
391 n_minor_intervals = num_minor_intervals(axis)
392 minorticks = sizehint!(eltype(ticks)[], n_minor_intervals * sub * length(ticks))
393 for i in 2:length(ticks)
394 lo = ticks[i - 1]
395 hi = ticks[i]
396 (isfinite(lo) && isfinite(hi) && hi > lo) || continue
397 if log_scaled
398 for e in 1:sub
399 lo_ = lo * base^(e - 1)
400 hi_ = lo_ * base
401 step = (hi_ - lo_) / n_minor_intervals
402 rng = (lo_ + (e > 1 ? 0 : step)):step:(hi_ - (e < sub ? 0 : step / 2))
403 append!(minorticks, collect(rng))
404 end
405 else
406 step = (hi - lo) / n_minor_intervals
407 append!(minorticks, collect((lo + step):step:(hi - step / 2)))
408 end
409 end
410 minorticks[amin . minorticks . amax]
411 end

626 function round_limits(amin, amax, scale)

628 factor = base^(1 - round(log(base, amax - amin)))
629 amin = floor(amin * factor) / factor
630 amax = ceil(amax * factor) / factor
631 amin, amax
632 end

636 process_limits(lims::Symbol, axis) = lims
637 process_limits(lims::AVec, axis) =
638 length(lims) == 2 && all(map(x -> x isa Union{Symbol,Real}, lims)) ? Tuple(lims) :
639 nothing
640 process_limits(lims, axis) = nothing

647 # using the axis extrema and limit overrides, return the min/max value for this axis
648 function axis_limits(
649 sp,
650 letter,
651 lims_factor = widen_factor(get_axis(sp, letter)),
652 consider_aspect = true,
653 )
654 axis = get_axis(sp, letter)
655 ex = axis[:extrema]
656 amin, amax = ex.emin, ex.emax
657 lims = process_limits(axis[:lims], axis)
658 lims === nothing && warn_invalid_limits(axis[:lims], letter)
659
660 if (has_user_lims = lims isa Tuple)
661 lmin, lmax = lims
662 if lmin isa Number && isfinite(lmin)
663 amin = lmin
664 elseif lmin isa Symbol
665 lmin === :auto || @warn "Invalid min $(letter)limit" lmin
666 end
667 if lmax isa Number && isfinite(lmax)
668 amax = lmax
669 elseif lmax isa Symbol
670 lmax === :auto || @warn "Invalid max $(letter)limit" lmax
671 end
672 end
673 if lims === :symmetric
674 amax = max(abs(amin), abs(amax))
675 amin = -amax
676 end
677 if amax amin && isfinite(amin)
678 amax = amin + 1.0
679 end
680 if !isfinite(amin) && !isfinite(amax)
681 amin, amax = zero(amin), one(amax)
682 end
683 if ispolar(axis.sps[1])
684 if axis[:letter] === :x
685 amin, amax = 0, 2
686 elseif lims === :auto
687 # widen max radius so ticks dont overlap with theta axis
688 amin, amax = 0, amax + 0.1abs(amax - amin)
689 end
690 elseif lims_factor !== nothing
691 amin, amax = scale_lims(amin, amax, lims_factor, axis[:scale])
692 elseif lims === :round
693 amin, amax = round_limits(amin, amax, axis[:scale])
694 end
695
696 aspect_ratio = get_aspect_ratio(sp)
697 if (
698 !has_user_lims &&
699 consider_aspect &&
700 letter in (:x, :y) &&
701 !(aspect_ratio === :none || RecipesPipeline.is3d(:sp))
702 )
703 aspect_ratio = aspect_ratio isa Number ? aspect_ratio : 1
704 area = plotarea(sp)
705 plot_ratio = height(area) / width(area)
706 dist = amax - amin
707
708 factor = if letter === :x
709 ydist, = axis_limits(sp, :y, widen_factor(sp[:yaxis]), false) |> collect |> diff
710 axis_ratio = aspect_ratio * ydist / dist
711 axis_ratio / plot_ratio
712 else
713 xdist, = axis_limits(sp, :x, widen_factor(sp[:xaxis]), false) |> collect |> diff
714 axis_ratio = aspect_ratio * dist / xdist
715 plot_ratio / axis_ratio
716 end
717
718 if factor > 1
719 center = (amin + amax) / 2
720 amin = center + factor * (amin - center)
721 amax = center + factor * (amax - center)
722 end
723 end
724
725 amin, amax
726 end

730 # these methods track the discrete (categorical) values which correspond to axis continuous values (cv)
731 # whenever we have discrete values, we automatically set the ticks to match.
732 # we return (continuous_value, discrete_index)
733 discrete_value!(plotattributes, letter::Symbol, dv) =
734 let l = if plotattributes[:permute] !== :none
735 filter(!=(letter), plotattributes[:permute]) |> only
736 else
737 letter
738 end
739 discrete_value!(plotattributes[:subplot][get_attr_symbol(l, :axis)], dv)
740 end
741
742 discrete_value!(axis::Axis, dv) =
743 if (cv_idx = get(axis[:discrete_map], dv, -1)) == -1
744 ex = axis[:extrema]
745 cv = NaNMath.max(0.5, ex.emax + 1)
746 expand_extrema!(axis, cv)
747 push!(axis[:discrete_values], dv)
748 push!(axis[:continuous_values], cv)
749 cv_idx = length(axis[:discrete_values])
750 axis[:discrete_map][dv] = cv_idx
751 cv, cv_idx
752 else
753 cv = axis[:continuous_values][cv_idx]
754 cv, cv_idx
755 end

757 # continuous value... just pass back with axis negative index
758 discrete_value!(axis::Axis, cv::Number) = (cv, -1)

760 # add the discrete value for each item. return the continuous values and the indices
761 function discrete_value!(axis::Axis, v::AVec)
762 cvec = zeros(axes(v))
763 discrete_indices = similar(Array{Int}, axes(v))
764 for i in eachindex(v)
765 cvec[i], discrete_indices[i] = discrete_value!(axis, v[i])
766 end
767 cvec, discrete_indices
768 end
