238 function __heatmap_edges(v::AVec, isedges::Bool, ispolar::Bool)
239 (n = length(v)) == 1 && return v[1] .+ [ispolar ? max(-v[1], -0.5) : -0.5, 0.5]
240 isedges && return v
241 # `isedges = true` means that v is a vector which already describes edges
242 # and does not need to be extended.
243 vmin, vmax = ignorenan_extrema(v)
244 extra_min = ispolar ? min(v[1], 0.5(v[2] - v[1])) : 0.5(v[2] - v[1])
245 extra_max = 0.5(v[n] - v[n - 1])
246 vcat(vmin - extra_min, 0.5(v[1:(n - 1)] + v[2:n]), vmax + extra_max)
247 end

634 """
635 Allows temporary setting of backend and defaults for Plots. Settings apply only for the `do` block. Example:
636 ```
637 Plots.with(:gr, size=(400,400), type=:histogram) do
638 plot(rand(10))
639 plot(rand(10))
640 end
641 ```
642 """
643 function with(f::Function, args...; scalefonts = nothing, kw...)
644 newdefs = KW(kw)
645
646 if :canvas in args
647 newdefs[:xticks] = nothing
648 newdefs[:yticks] = nothing
649 newdefs[:grid] = false
650 newdefs[:legend_position] = false
651 end
652
653 # dict to store old and new keyword args for anything that changes
654 olddefs = KW()
655 for k in keys(newdefs)
656 olddefs[k] = default(k)
657 end
658
659 # save the backend
660 CURRENT_BACKEND.sym === :none && _pick_default_backend()
661 oldbackend = CURRENT_BACKEND.sym
662
663 for arg in args
664 # change backend?
665 arg in backends() && backend(arg)
666
667 # TODO: generalize this strategy to allow args as much as possible
668 # as in: with(:gr, :scatter, :legend, :grid) do; ...; end
669 # TODO: can we generalize this enough to also do something similar in the plot commands??
670
671 k = :legend
672 if arg in (k, :leg)
673 olddefs[k] = default(k)
674 newdefs[k] = true
675 end
676
677 k = :grid
678 if arg == k
679 olddefs[k] = default(k)
680 newdefs[k] = true
681 end
682 end
683
684 # now set all those defaults
685 default(; newdefs...)
686 scalefonts ≡ nothing || scalefontsizes(scalefonts)
687
688 # call the function
689 ret = f()
690
691 # put the defaults back
692 scalefonts ≡ nothing || resetfontsizes()
693 default(; olddefs...)
694
695 # revert the backend
696 CURRENT_BACKEND.sym != oldbackend && backend(oldbackend)
697
698 # return the result of the function
699 ret
700 end

973 function ___straightline_data(xl, yl, x, y, exp_fact)
974 x_vals, y_vals = if y[1] == y[2]
975 if x[1] == x[2]
976 error("Two identical points cannot be used to describe a straight line.")
977 else
978 [xl[1], xl[2]], [y[1], y[2]]
979 end
980 elseif x[1] == x[2]
981 [x[1], x[2]], [yl[1], yl[2]]
982 else
983 # get a and b from the line y = a * x + b through the points given by
984 # the coordinates x and x
985 b = y[1] - (y[1] - y[2]) * x[1] / (x[1] - x[2])
986 a = (y[1] - y[2]) / (x[1] - x[2])
987 # get the data values
988 xdata = [
989 clamp(x[1] + (x[1] - x[2]) * (ylim - y[1]) / (y[1] - y[2]), xl...) for
990 ylim in yl
991 ]
992
993 xdata, a .* xdata .+ b
994 end
995 # expand the data outside the axis limits, by a certain factor too improve
996 # plotly(js) and interactive behaviour
997 (
998 x_vals .+ (x_vals[2] - x_vals[1]) .* exp_fact,
999 y_vals .+ (y_vals[2] - y_vals[1]) .* exp_fact,
1000 )
1001 end
1002
1003 __straightline_data(xl, yl, x, y, exp_fact) =
1004 if (n = length(x)) == 2
1005 ___straightline_data(xl, yl, x, y, exp_fact)
1006 else
1007 k, r = divrem(n, 3)
1008 @assert r == 0 "Misformed data. `straightline_data` either accepts vectors of length 2 or 3k. The provided series has length $n"
1009 xdata, ydata = fill(NaN, n), fill(NaN, n)
1010 for i in 1:k
1011 inds = (3i - 2):(3i - 1)
1012 xdata[inds], ydata[inds] =
1013 ___straightline_data(xl, yl, x[inds], y[inds], exp_fact)
1014 end
1015 xdata, ydata
1016 end

1190 # The following functions implement the guess of the optimal legend position,
1191 # from the data series.
1192 function d_point(x, y, lim, scale)
1193 p_scaled = (x / scale[1], y / scale[2])
1194 d = sum(abs2, lim .- p_scaled)
1195 isnan(d) && return 0.0
1196 d
1197 end
1198 # Function barrier because lims are type-unstable
1199 function _guess_best_legend_position(xl, yl, plt, weight = 100)
1200 scale = (maximum(xl) - minimum(xl), maximum(yl) - minimum(yl))
1201 u = zeros(4) # faster than tuple
1202 # Quadrants where the points will be tested
1203 quadrants = (
1204 ((0.00, 0.25), (0.00, 0.25)), # bottomleft
1205 ((0.75, 1.00), (0.00, 0.25)), # bottomright
1206 ((0.00, 0.25), (0.75, 1.00)), # topleft
1207 ((0.75, 1.00), (0.75, 1.00)), # topright
1208 )
1209 for series in plt.series_list
1210 x = series[:x]
1211 y = series[:y]
1212 yoffset = firstindex(y) - firstindex(x)
1213 for (i, lim) in enumerate(Iterators.product(xl, yl))
1214 lim = lim ./ scale
1215 for ix in eachindex(x)
1216 xi, yi = x[ix], _cycle(y, ix + yoffset)
1217 # ignore y points outside quadrant visible quadrant
1218 xi < xl[1] + quadrants[i][1][1] * (xl[2] - xl[1]) && continue
1219 xi > xl[1] + quadrants[i][1][2] * (xl[2] - xl[1]) && continue
1220 yi < yl[1] + quadrants[i][2][1] * (yl[2] - yl[1]) && continue
1221 yi > yl[1] + quadrants[i][2][2] * (yl[2] - yl[1]) && continue
1222 u[i] += inv(1 + weight * d_point(xi, yi, lim, scale))
1223 end
1224 end
1225 end
1226 # return in the preferred order in case of draws
1227 ibest = findmin(u)[2]
1228 u[ibest] ≈ u[4] && return :topright
