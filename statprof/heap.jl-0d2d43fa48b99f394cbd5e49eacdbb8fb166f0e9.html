7 # Binary heap indexing
8 heapleft(i::Integer) = 2i
9 heapright(i::Integer) = 2i + 1
10 heapparent(i::Integer) = div(i, 2)

13 # Binary min-heap percolate down.
14 function percolate_down!(xs::Vector, i::Integer, x, o::Ordering, len::Integer=length(xs))
15 @inbounds while (l = heapleft(i)) <= len
16 r = heapright(i)
17 j = r > len || lt(o, xs[l], xs[r]) ? l : r
18 lt(o, xs[j], x) || break
19 xs[i] = xs[j]
20 i = j
21 end
22 xs[i] = x
23 end

25 # Binary min-heap percolate up.
26 function percolate_up!(xs::Vector, i::Integer, x, o::Ordering)
27 @inbounds while (j = heapparent(i)) >= 1
28 lt(o, x, xs[j]) || break
29 xs[i] = xs[j]
30 i = j
31 end
32 xs[i] = x
33 end

36 heappop!(v, ord)

38 Given a binary heap-ordered array, remove and return the lowest ordered element.
39 For efficiency, this function does not check that the array is indeed heap-ordered.

41 1 (0 %) 1 (0 %) function heappop!(xs::Vector, o::Ordering)
42 x = xs[1]
43 y = pop!(xs)
44 if !isempty(xs)
45 percolate_down!(xs, 1, y, o)

52
53 Given a binary heap-ordered array, push a new element `x`, preserving the heap property.

56 function heappush!(xs::Vector, x, o::Ordering)
57 push!(xs, x)
58 i = lastindex(xs)
59 percolate_up!(xs, i, @inbounds(xs[i]), o)

67 Turn an arbitrary vector into a binary min-heap in linear time.

69 function heapify!(xs::Vector, o::Ordering)
70 for i in heapparent(lastindex(xs)):-1:1
71 percolate_down!(xs, i, @inbounds(xs[i]), o)
